module.exports = async function handler(req, res) { const token = process.env.TELEGRAM_BOT_TOKEN; const chatId = process.env.TELEGRAM_CHAT_ID; if (!token) return res.status(200).json({ ok:false, tokenConfigured:false, chatIdConfigured:!!chatId, error:'TELEGRAM_BOT_TOKEN missing' }); const r = await fetch(`https://api.telegram.org/bot${token}/getMe`); const data = await r.json().catch(() => ({})); res.status(200).json({ ok: !!data.ok && !!chatId, tokenConfigured:true, chatIdConfigured:!!chatId, telegramOk:!!data.ok, bot:data.result?.username || null, telegramError:data.description || null }); }
